# Fun√ß√£o para obter informa√ß√µes do arquivo
get_file_info() {
    local file="$1"
    
    if [[ ! -f "$file" ]]; then
        echo "0|unknown|unknown"
        return
    fi
    
    local size=$(stat -c%s "$file" 2>/dev/null || echo "0")
    local timestamp=$(stat -c%Y "$file" 2>/dev/null || echo "0")
    local readable_size=$(numfmt --to=iec-i --suffix=B "$size" 2>/dev/null || echo "0B")
    local readable_date=$(date -d "@$timestamp" '+%Y-%m-%d %H:%M:%S' 2>/dev/null || echo "unknown")
    
    echo "$readable_size|$readable_date|$timestamp"
}

# Fun√ß√£o para verificar duplicatas
check_duplicate() {

            echo "identical"
        else
            echo "different"
        fi
    else
        echo "new"
    fi
}

# Fun√ß√£o para gerar nome √∫nico
generate_unique_name() {
    local dest_dir="$1"
    local filename="$2"
    local base_name="${filename%.*}"
    local extension="${filename##*.}"
    local counter=1
    
    # Se n√£o tem extens√£o

    fi
    
    local new_name="$filename"
    while [[ -f "$dest_dir/$new_name" ]]; do
        if [[ -n "$extension" ]]; then
            new_name="${base_name}_${counter}.${extension}"
        else
            new_name="${base_name}_${counter}"
        fi
        ((counter++))
    done
    
    echo "$new_name"
}

                    fi
                    ;;
                "new")
                    if [[ "$DRY_RUN" == "false" ]]; then

                            ((error_files++))
                        fi

                        log "INFO" "[DRY RUN] Moveria: $filename -> $file_type/$file_year/$file_month/"
                    fi
                    ;;
            esac
            
            ((processed_files++))
            
            # Mostra progresso

        fi
    done < <(find "$DOWNLOADS_DIR" -type f -not -path "*/.*" -print0)
    
    echo  # Nova linha ap√≥s barra de progresso
    
    # Relat√≥rio final

# Fun√ß√£o para gerar relat√≥rio detalhado
generate_report() {
    local report_file="$TARGET_DIR/organization_report_$(date +%Y%m%d_%H%M%S).txt"
    
    log "INFO" "Gerando relat√≥rio detalhado..."
    
    {
        echo "=== RELAT√ìRIO DE ORGANIZA√á√ÉO DE ARQUIVOS ==="
        echo "Data: $(date)"
        echo "Diret√≥rio origem: $DOWNLOADS_DIR"
        echo "Diret√≥rio destino: $TARGET_DIR"
        echo
        
        echo "=== ESTAT√çSTICAS POR TIPO ==="
        for type in images documents videos audio code archives executables other; do
            if [[ -d "$TARGET_DIR/$type" ]]; then
                local count=$(find "$TARGET_DIR/$type" -type f 2>/dev/null | wc -l)
                local size=$(du -sh "$TARGET_DIR/$type" 2>/dev/null | cut -f1)
                printf "%-12s: %6d arquivos (%s)\n" "$type" "$count" "$size"
            fi
        done
        
        echo
        echo "=== DISTRIBUI√á√ÉO POR ANO ==="
        for year_dir in "$TARGET_DIR"/*/*; do

                local size=$(du -sh "$year_dir" 2>/dev/null | cut -f1)
                printf "%-12s/%s: %6d arquivos (%s)\n" "$type" "$year" "$count" "$size"
            fi
        done
        
        echo
        echo "=== ARQUIVOS MAIORES ==="
        echo "Top 10 maiores arquivos:"
        find "$TARGET_DIR" -type f -exec ls -lh {} \; 2>/dev/null | \
            sort -k5 -hr | head -10 | \

    } > "$report_file"
